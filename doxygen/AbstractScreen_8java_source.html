58  public char charAt(final int x, final int y) {
59  if (x < 0 || x >= width || y < 0 || y >= height) {
60  throw new IndexOutOfBoundsException("(" + x + ", " + y + ")" + ", should be in (0.." + width + ", 0.."
61  + height + ")");
62  }
63  final InputField f = getInputFieldAt(x, y);
64  if (f != null) {
65  final String value = f.getValue();
66  return value.charAt(x - f.getStartX());
67  } else {
68  final char[] line = buffer[y];
