206  public Result doCommand(final String command) {
207  assertConnected();
208  try {
209  out.println(command);
210  out.flush();
211  if (logger.isDebugEnabled()) {
212  logger.debug("---> " + command);
213  }
214 
215  // new s3270 terminal > 3.5 needs a short time.
216  // @TODO: needs to be analyzed
217  try {
218  Thread.sleep(100L);
219  } catch (InterruptedException e) {
220  logger.error("i got interrupted.", e);
221  }
222 
223  final List<String> lines = new ArrayList<String>();
224  while (true) {
225  final String line = in.readLine();
226  if (line == null) {
227  checkS3270Process(); // will throw appropriate exception
228  // if we get here, it's a more obscure error
229  throw new RuntimeException("s3270 process not responding");
230  }
231 
232  if (logger.isDebugEnabled()) {
233  logger.debug("<--- " + line);
234  }
235 
236  if (line.equals("ok")) {
237  break;
238  }
239  lines.add(line);
240  }
241  final int size = lines.size();
242  if (size > 0) {
243  return new Result(lines.subList(0, size - 1), (String) lines.get(size - 1));
244  } else {
245  throw new RuntimeException("no status received in command: " + command);
246  }
247  } catch (final IOException ex) {
248  throw new RuntimeException("IOException during command: " + command, ex);
249  }
250  }

276  private static final Pattern unknownHostPattern = Pattern.compile(
277  // This message is hard-coded in s3270 as of version 3.3.5,
278  // so we can rely on it not being localized.
279  "Unknown host: (.*)");
280  private static final Pattern unreachablePattern = Pattern.compile(
281  // This is the hard-coded part of the error message in s3270 version 3.3.5.
282  "Connect to ([^,]+), port ([0-9]+): (.*)");
283 
288  private void checkS3270Process() {
289  // Ideally, we'd like to call Process.waitFor() with a timeout,
290  // but that is so complicated to implement that we take a
291  // second-rate approach: wait a little while, and then check if
292  // the process is already terminated.
293  try {
294  Thread.sleep(100);
295  } catch (final InterruptedException ex) {
296  }
297  try {
298  final int exitValue = s3270.exitValue();
299  final String message = errorReader.message;
300  if (exitValue == 1 && message != null) {
301  Matcher m = unknownHostPattern.matcher(message);
302  if (m.matches()) {
303  throw new UnknownHostException(m.group(1));
304  } else {
305  m = unreachablePattern.matcher(message);
306  if (m.matches()) {
307  throw new HostUnreachableException(m.group(1), m.group(3));
308  }
309  }
310  throw new S3270Exception("s3270 terminated with code " + exitValue + ", message: "
311  + errorReader.message);
312  }
313  } catch (final IllegalThreadStateException ex) {
314  // we get here if the process has still been running in the
315  // call to s3270.exitValue() above
316  throw new S3270Exception("s3270 not terminated, error: " + errorReader.message);
317  }
318  }

420  public void submitScreen() {
421  assertConnected();
422  for (final Iterator<Field> i = screen.getFields().iterator(); i.hasNext();) {
423  final Field f = i.next();
424  if ((f instanceof InputField) && ((InputField) f).isChanged()) {
425  doCommand("movecursor (" + f.getStartY() + ", " + f.getStartX() + ")");
426  doCommand("eraseeof");
427  final String value = f.getValue();
428  for (int j = 0; j < value.length(); j++) {
429  final char ch = value.charAt(j);
430  if (ch == '\n') {
431  doCommand("newline");
432  } else if (!Integer.toHexString(ch).equals("0")) {
433  doCommand("key (0x" + Integer.toHexString(ch) + ")");
434  }
435  }

504  private static final Pattern FUNCTION_KEY_PATTERN = Pattern.compile("p(f|a)([0-9]{1,2})");
505 
506  @SuppressWarnings("unchecked")
507  public void doKey(final String key) {
508  assertConnected();
509  final Matcher m = FUNCTION_KEY_PATTERN.matcher(key);
510  if (m.matches()) { // function key
511  final int number = Integer.parseInt(m.group(2));
512  if (m.group(1).equals("f")) {
513  this.pf(number);
514  } else {
515  this.pa(number);
516  }
517  } else if (key.equals("")) {
518  // use ENTER as a default action if the actual key got lost
519  this.enter();
520  } else { // other key: find a parameterless method of the same name
521  try {
522  final Class c = this.getClass();
523  final Method method = c.getMethod(key, new Class[] {});
524  method.invoke(this, new Object[] {});
525  } catch (final NoSuchMethodException ex) {
526  throw new IllegalArgumentException("no such key: " + key);
527  } catch (final IllegalAccessException ex) {
528  throw new RuntimeException("illegal s3270 method access for key: " + key);
529  } catch (final InvocationTargetException ex) {
530  throw new RuntimeException("error invoking s3270 for key: " + key + ", exception: "
531  + ex.getTargetException());
532  }
533  }
534  }
